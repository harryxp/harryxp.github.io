Monad Laws

  • Identity

      (return x) >>= f      ≡ f x
  m          >>= return ≡ m

  return     >=> f      ≡ f
  f          >=> return ≡ f

  • Associativity

      (m >>= f)  >>= g      ≡ m >>= (\x -> f x >>= g)

  (f >=> g)  >=> h      ≡ f >=> (g >=> h)

Power set:

import Control.Monad (filterM)
powerSet :: [a] -> [[a]]
powerSet =  filterM (\_ -> [True,False])

All possible totals from a set of coins.

import Control.Monad (foldM)
totals :: [Int] -> [Int]
totals = foldM (\accum coin -> [accum, accum + coin]) 0
